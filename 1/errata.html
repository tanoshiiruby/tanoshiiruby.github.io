class Vector
  :
  def +@
    self.dup               # 自分の複製を返す
  end
  def -@
    Vector.new(-@x, -@y)   # x, y のそれぞれの正負を逆にする
  end
  def ~@
    Vector.new(-@y, @x)    # 90度反転させた座標を返す
  end
end

vec = Vector.new(3, 6)
p +vec          #=> (3, 6)
p -vec          #=> (-3, -6)
p ~vec          #=> (-6, 3)
